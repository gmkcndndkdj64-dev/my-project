  <input type="text" id="id">

  <label for="wallet">رقم المحفظة</label>
  <input type="text" id="wallet">

  <button onclick="addOwner()">تسجيل المالك</button>

  <hr>

  <label for="search">بحث (بالاسم أو رقم البطاقة أو رقم المحفظة)</label>
  <input type="text" id="search" oninput="searchOwners()">

  <div class="result" id="results"></div>
</div>

<script>
  let owners = JSON.parse(localStorage.getItem('owners')) || [];

  function saveOwners() {
    localStorage.setItem('owners', JSON.stringify(owners));
  }

  function addOwner() {
    const name = document.getElementById('name').value.trim();
    const id = document.getElementById('id').value.trim();
    const wallet = document.getElementById('wallet').value.trim();

    if (!name || !id || !wallet) {
      alert('الرجاء تعبئة جميع الحقول.');
      return;
    }

    owners.push({ name, id, wallet });
    saveOwners();
    clearFields();
    searchOwners(); // عرض البيانات بعد التحديث
  }

  function clearFields() {
    document.getElementById('name').value = '';
    document.getElementById('id').value = '';
    document.getElementById('wallet').value = '';
  }

  function searchOwners() {
    const query = document.getElementById('search').value.toLowerCase();
    const resultDiv = document.getElementById('results');
    resultDiv.innerHTML = '';

    const filtered = owners.filter(o =>
      o.name.toLowerCase().includes(query) ||
      o.id.includes(query) ||
      o.wallet.includes(query)
    );

    if (filtered.length === 0) {
      resultDiv.innerHTML = '<p>لا توجد نتائج.</p>';
      return;
    }

    filtered.forEach(o => {
      const entry = document.createElement('div');
      entry.className = 'entry';
      entry.innerHTML = `<strong>الاسم:</strong> ${o.name}<br>
                         <strong>رقم البطاقة:</strong> ${o.id}<br>
                         <strong>رقم المحفظة:</strong> ${o.wallet}`;
      resultDiv.appendChild(entry);
    });
  }

  // عرض كل البيانات عند أول تحميل
  window.onload = searchOwners;
</script>
